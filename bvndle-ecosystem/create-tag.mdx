```bash cURL

'https://test-api.bvndle.com/api/v1/ufel/users/create-user-tag' \
--header 'Content-Type: application/json' \

{
    "epPartnerId": "Sample 32 no string",
    "epPartnerUserId": "Sample 32 no string",

    "password": "Sample password",
    "email": "test@mail.com",
}
```
