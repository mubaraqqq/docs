###

```bash cURL
curl --request PUT
'https://test-api.bvndle.com/api/v1/laas/spin-wheel/3a035dee-8ad5-4fed-bcec-91261dda2272/prizes/5707d43c-9270-4137-9a4a-d5a49187bc18' \
--header 'Content-Type: application/json' \

{
    "coin": 2,
    "title": "2 coins",
    "probability": 50,
    "type": "User Credit",
    "isActive": false
}

```
